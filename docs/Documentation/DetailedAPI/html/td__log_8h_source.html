23/**
24 * Sets the path to the file where the internal TDLib log will be written.
25 * By default TDLib writes logs to stderr or an OS specific log.
26 * Use this method to write the log to a file instead.
27 *
28 * \deprecated Use synchronous setLogStream request instead.
29 * \param[in] file_path Null-terminated path to a file where the internal TDLib log will be written.
30 * Use an empty path to switch back to the default logging behaviour.
31 * \return True 1 on success, or 0 otherwise, i.e. if the file can't be opened for writing.
32 */
33TDJSON_DEPRECATED_EXPORT int td_set_log_file_path(const char *file_path);
34
35/**
36 * Sets the maximum size of the file to where the internal TDLib log is written before the file will be auto-rotated.
37 * Unused if log is not written to a file. Defaults to 10 MB.
38 *
39 * \deprecated Use synchronous setLogStream request instead.
40 * \param[in] max_file_size The maximum size of the file to where the internal TDLib log is written before the file
41 * will be auto-rotated. Should be positive.
42 */
43TDJSON_DEPRECATED_EXPORT void td_set_log_max_file_size(long long max_file_size);
44
45/**
46 * Sets the verbosity level of the internal logging of TDLib.
47 * By default the TDLib uses a log verbosity level of 5.
48 *
49 * \deprecated Use synchronous setLogVerbosityLevel request instead.
50 * \param[in] new_verbosity_level New value of logging verbosity level.
51 * Value 0 corresponds to fatal errors,
52 * value 1 corresponds to errors,
53 * value 2 corresponds to warnings and debug warnings,
54 * value 3 corresponds to informational,
55 * value 4 corresponds to debug,
56 * value 5 corresponds to verbose debug,
57 * value greater than 5 and up to 1024 can be used to enable even more logging.
58 */
59TDJSON_DEPRECATED_EXPORT void td_set_log_verbosity_level(int new_verbosity_level);

61/**
62 * A type of callback function that will be called when a fatal error happens.
63 *
64 * \param error_message Null-terminated string with a description of a happened fatal error.
65 */
66typedef void (*td_log_fatal_error_callback_ptr)(const char *error_message);

68/**
69 * Sets the callback that will be called when a fatal error happens.
70 * None of the TDLib methods can be called from the callback.
71 * The TDLib will crash as soon as callback returns.
72 * By default the callback is not set.
73 *
74 * \deprecated Use td_set_log_message_callback instead.
75 * \param[in] callback Callback that will be called when a fatal error happens.
76 * Pass NULL to remove the callback.
77 */
78TDJSON_DEPRECATED_EXPORT void td_set_log_fatal_error_callback(td_log_fatal_error_callback_ptr callback);
